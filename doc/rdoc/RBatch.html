def common_config
  if File.exist?(RBatch.common_config_path)
    yaml = YAML::load_file(RBatch.common_config_path)
    if yaml
      return yaml
    else
      # If file is emply , YAML::load_file is false
      return nil
    end
  else
    return nil
  end
end

def tmp_dir
  case RUBY_PLATFORM
  when /mswin|mingw/
    return ENV["TEMP"]
  when /cygwin|linux/
    return "/tmp/"
  else
    raise "Unknown RUBY_PRATFORM : " + RUBY_PLATFORM
  end
end
